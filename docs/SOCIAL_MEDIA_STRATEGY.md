(function(i,s,o,g,r,a,m){
    i['GoogleAnalyticsObject']=r;
    i[r]=i[r]||function(){
        (i[r].q=i[r].q||[]).push(arguments);
    };
    i[r].l=1*new Date();

ga('create', 'UA-XXXXX-Y', 'auto');

// Track Voting
function trackVote(voteType, productId) {
    ga('send', 'event', {
        'eventCategory': 'Voting',
        'eventAction': voteType,
        'eventLabel': productId,
        'value': 1
    });
}

// Track Amazon Click
function trackAmazonClick(url) {
    ga('send', 'event', {
        'eventCategory': 'Affiliate',
        'eventAction': 'click',
        'eventLabel': url,
        'value': 1
    });
}

## 🤝 Community Building

### Contributor Program

**Levels:**
1. **Contributor** - Issues, Code
2. **Maintainer** - Review, Merge
3. **Owner** - Full Access

**Benefits:**
- Discord Role
- GitHub Badge
- Early Access
- Recognition

### Community Guidelines

1. **Sei respektvoll** - Keine Hate Speech
2. **Konstruktiv** - Konstruktive Kritik
3. **Datenschutz** - Keine PII teilen
4. **Lizenz** - MIT für alle Beiträge

### Engagement Tactics

- **Beantworten** alle GitHub Issues (24h)
- **Review** PRs innerhalb 48h
- **Feature-Requests** priorisieren
- **Community-Posts** fördern
- **Events** organisieren (Hackathons)
